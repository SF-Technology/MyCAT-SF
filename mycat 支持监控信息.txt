DROP TABLE IF EXISTS t_memory;
CREATE TABLE t_memory(thread_id INT PRIMARY KEY,thread_name VARCHAR(255),memory_type VARCHAR(255),used BIGINT,free BIGINT,total BIGINT);
INSERT INTO t_memory VALUES(1, 'Hello','dd',12,1,22);
INSERT INTO t_memory VALUES(2, 'World','dd',12,1,22);
SELECT * FROM t_memory ORDER BY thread_id;
UPDATE t_memory SET thread_name ='Hi' WHERE thread_id=1;
DELETE FROM t_memory WHERE thread_id=2;

	name

DROP TABLE IF EXISTS 	t_threadpool;
CREATE TABLE 	t_threadpool(thread_name VARCHAR(255) PRIMARY KEY,pool_size BIGINT,active_count INT,task_queue_size INT,completed_task INT,total_task INT);
INSERT INTO t_threadpool VALUES('Hello',12,1,22,22,22);
INSERT INTO t_threadpool VALUES('World',12,1,22,22,222);
SELECT * FROM t_threadpool ORDER BY thread_name;
UPDATE t_threadpool SET pool_size = 1100 WHERE thread_name='Hello';
DELETE FROM t_threadpool WHERE thread_name='World';

3.数据库连接池(后端数据库连接)
	t_connectpool
	
DROP TABLE IF EXISTS 	t_connectpool;
CREATE TABLE 	t_connectpool(processor VARCHAR(255) PRIMARY KEY,
								id INT,mysqlId INT,
								host VARCHAR(32),port INT,l_port INT,
								net_in BIGINT,net_out BIGINT,
								life BIGINT,closed VARCHAR(16),
								borrowed VARCHAR(16),SEND_QUEUE INT,
								schema VARCHAR(32),charset VARCHAR(32),
								txlevel VARCHAR(16),autocommit VARCHAR(16));



INSERT INTO t_connectpool VALUES('Hello1',12,1,'www.baidu.com',33,22,100000,100000,200,'false','false',112,'db_1','utf8','1','0');
